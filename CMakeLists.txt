if(UNIX AND NOT APPLE)
    cmake_minimum_required(VERSION 3.23.0)
    project(scarab LANGUAGES CXX)

    if(NOT FLATPAK)
        if(EXISTS ${CMAKE_CURRENT_BINARY_DIR}/gcem)
            message("Using existing gcem clone")
        else()
            execute_process(COMMAND git clone --depth=1 https://github.com/kthohr/gcem)
        endif()

        if(EXISTS ${CMAKE_CURRENT_BINARY_DIR}/OD_components)
            message("Using existing OD components")
        else()
            execute_process(COMMAND curl https://datashare.mpcdf.mpg.de/public.php/dav/files/rpxPyMEQrWqwKBW --output OD_components.tar.gz)
            execute_process(COMMAND tar -xzf OD_components.tar.gz)
        endif()
    endif()

    find_package(PkgConfig REQUIRED)
    find_package(fmt REQUIRED)
    if(FLATPAK)
        include_directories(/app/include)
        link_directories(/app/lib)
    else()
        include_directories(${CMAKE_CURRENT_BINARY_DIR}/OD_components/include)
        include_directories(${CMAKE_CURRENT_BINARY_DIR}/gcem/include)
        include_directories(/usr/local/OceanOptics/OceanDirect/include)
        link_directories(/usr/local/OceanOptics/OceanDirect/lib)
        link_directories(${CMAKE_CURRENT_BINARY_DIR}/OD_components/lib)
    endif()

    pkg_check_modules(GTK4 gtk4)
    include_directories(${GTK4_INCLUDE_DIRS})


    link_directories(${GTK4_LIBRARY_DIRS})

    add_compile_options(-std=c++20 -O3 -fopenmp -w -Wall)
    add_executable(scarab
        Scarab/Scarab.cpp)
    target_link_libraries(scarab ${GTK4_LIBRARIES})
    target_link_libraries(scarab fmt::fmt -lm -loceandirect)
    target_link_libraries(scarab -lgomp -lpthread -ldl)
    install(TARGETS scarab)
endif()
